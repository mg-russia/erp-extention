
Процедура ПередЗаписью(Отказ)
	
	Ур =  Уровень();
	Если Ур = 0 Тогда
		Поиск = Справочники._Типы.НайтиПоКоду(СокрЛП(Код));
		Наименование = Поиск.Имя;
	ИначеЕсли Ур = 1 Тогда
		Поиск = Справочники._Модели.НайтиПоКоду(СокрЛП(Код));
		Если Не Поиск.Имя = "" Тогда 
			Наименование = Поиск.Имя;
		Иначе
			Поиск = Справочники._Модели.НайтиПоКоду(СокрЛП(Родитель.Код));
			Наименование = Поиск.Имя;
		КонецЕсли;	
	ИначеЕсли Ур = 2 Тогда
		Поиск = Справочники._Измерения.НайтиПоКоду(СокрЛП(Код));
		Если Не Поиск.Имя = "" И Не Поиск.Наименование = "-"  Тогда 
			Наименование = Поиск.Имя;
		Иначе
			Поиск = Справочники._Модели.НайтиПоКоду(СокрЛП(Родитель.Код));
			Наименование = Поиск.Имя + " -";
		КонецЕсли;
	КонецЕсли;
	
КонецПроцедуры
