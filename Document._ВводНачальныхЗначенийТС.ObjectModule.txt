
Процедура ОбработкаПроведения(Отказ, Режим)
	
	Для Каждого ТекСтрокаНачальныеЗначения Из НачальныеЗначения Цикл
		Если Не ЗначениеЗаполнено(ТекСтрокаНачальныеЗначения.ТС) Тогда Продолжить; КонецЕсли;
		Если Не ЗначениеЗаполнено(ТекСтрокаНачальныеЗначения.ПоказанияСпидометра) Тогда Продолжить; КонецЕсли;
		Движение = Движения._СчетчикиТС.Добавить();
		Движение.Период      = Дата;
		Движение.ТС          = ТекСтрокаНачальныеЗначения.ТС;
		Движение.ТипСчетчика = Перечисления._ТипыСчетчиковТС.Спидометр;
		Движение.Значение    = ТекСтрокаНачальныеЗначения.ПоказанияСпидометра;
	КонецЦикла;

	// регистр _ОстаткиГСМНаТС Приход
	Для Каждого ТекСтрокаНачальныеЗначения Из НачальныеЗначения Цикл
		Если Не ЗначениеЗаполнено(ТекСтрокаНачальныеЗначения.ТС) Тогда Продолжить; КонецЕсли;
		Если Не ЗначениеЗаполнено(ТекСтрокаНачальныеЗначения.ГСМ) Тогда Продолжить; КонецЕсли;
		Если Не ЗначениеЗаполнено(ТекСтрокаНачальныеЗначения.Количество) Тогда Продолжить; КонецЕсли;
		Движение             = Движения._ОстаткиГСМНаТС.Добавить();
		Движение.ВидДвижения = ВидДвиженияНакопления.Приход;
		Движение.Период      = Дата;
		Движение.ТС          = ТекСтрокаНачальныеЗначения.ТС;
		Движение.ГСМ         = ТекСтрокаНачальныеЗначения.ГСМ;
		Движение.Количество  = ТекСтрокаНачальныеЗначения.Количество;
	КонецЦикла;
